<script type="module">

  authDomain: "ldr-39d20.firebaseapp.com",
  projectId: "ldr-39d20",
  storageBucket: "ldr-39d20.firebasestorage.app",
  messagingSenderId: "832106604585",
  appId: "1:832106604585:web:8fe3c6d57fe1fd75df5438"
};

const app = initializeApp(firebaseConfig);
const db = getFirestore(app);

const docRef = doc(db, "ourspace", "shared");

let movies = [];
let questions = [];
let savings = 0;

/* REALTIME SYNC */
onSnapshot(docRef, (snap) => {
  if (snap.exists()) {
    const data = snap.data();
    movies = data.movies || [];
    questions = data.questions || [];
    savings = data.savings || 0;

    render();
  }
});

/* SAVE */
async function save() {
  await setDoc(docRef, {
    movies,
    questions,
    savings
  });
}

/* UI RENDER */
function render() {
  const mList = document.getElementById("movieList");
  const qList = document.getElementById("questionList");

  mList.innerHTML = movies.map(m => `<div class="tag">${m}</div>`).join("");
  qList.innerHTML = questions.map(q => `<div class="tag">${q}</div>`).join("");

  document.getElementById("total").innerText = "₹" + savings;
}

/* ACTIONS */
window.addMovie = () => {
  const val = movieInput.value.trim();
  if (!val) return;
  movies.push(val);
  movieInput.value = "";
  save();
};

window.addQuestion = () => {
  const val = questionInput.value.trim();
  if (!val) return;
  questions.push(val);
  questionInput.value = "";
  save();
};

window.addMoney = () => {
  const val = parseFloat(moneyInput.value);
  if (!val) return;
  savings += val;
  moneyInput.value = "";
  save();
};
